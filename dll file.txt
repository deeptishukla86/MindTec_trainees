using System;

namespace test1
{
    public class Class1
    {
        public void circle()
        {
            double r, a = 0.0;
            Console.WriteLine("Enter the radius of the circle:");
            r = Convert.ToDouble(Console.ReadLine());
            a = (3.14 * r * r);
            Console.WriteLine(" The area of the circle is : " + a);

        }
        public void rhombus()
        {
            double d1, d2, area = 0.0;
            Console.WriteLine("Enter the value of the diagonals :");
            d1 = Convert.ToDouble(Console.ReadLine());
            d2 = Convert.ToDouble(Console.ReadLine());
            area = (d1 * d2)/2;
            Console.WriteLine(" The area of the Rhombus is : " + area);
        }
    }
}
